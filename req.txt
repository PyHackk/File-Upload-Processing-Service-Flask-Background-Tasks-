// ============================================
// TABLEAU API CLASS
// ============================================
export class TableauAPI {
    private static readonly ENDPOINTS = {
        EXPLORE: '/api/tableau/insights/explore/',
        CATEGORIES: '/api/tableau/insights/categories/',
        WORKBOOK_DASHBOARDS: '/api/tableau/insights/workbook_dashboards/',
        RECENT: '/api/tableau/insights/recent/',
        POPULAR: '/api/tableau/insights/popular/',
        STATS: '/api/tableau/insights/stats/',
        SEARCH: '/api/tableau/insights/search/',
    };

    static async explore(search: string = '', limit: number = 100): Promise<any> {
        try {
            const response = await apiClient.get(
                this.ENDPOINTS.EXPLORE,
                { params: { search, limit } }
            );
            return response.data;
        } catch (error: any) {
            console.error('Explore API ERROR : ', error.response?.data || error.message);
            throw error;
        }
    }

    static async getCategories(): Promise<any> {
        try {
            const response = await apiClient.get(this.ENDPOINTS.CATEGORIES);
            return response.data;
        } catch (error: any) {
            console.error('Category API ERROR : ', error.response?.data || error.message);
            throw error;
        }
    }

    static async getWorkbookDashboards(WorkbookId: number): Promise<any> {
        try {
            const response = await apiClient.get(
                this.ENDPOINTS.WORKBOOK_DASHBOARDS,
                { params: { workbook_id: WorkbookId } }
            );
            return response.data;
        } catch (error: any) {
            console.error('Workbook API ERROR : ', error.response?.data || error.message);
            throw error;
        }
    }

    static async getRecent(limit: number = 10): Promise<any> {
        try {
            const response = await apiClient.get(
                this.ENDPOINTS.RECENT,
                { params: { limit } }
            );
            return response.data;
        } catch (error: any) {
            console.error('Recent API ERROR : ', error.response?.data || error.message);
            throw error;
        }
    }

    static async getPopular(limit: number = 10): Promise<any> {
        try {
            const response = await apiClient.get(
                this.ENDPOINTS.POPULAR,
                { params: { limit } }
            );
            return response.data;
        } catch (error: any) {
            console.error('Popular API ERROR : ', error.response?.data || error.message);
            throw error;
        }
    }

    static async getStats(): Promise<any> {
        try {
            const response = await apiClient.get(this.ENDPOINTS.STATS);
            return response.data;
        } catch (error: any) {
            console.error('Stats API ERROR : ', error.response?.data || error.message);
            throw error;
        }
    }

    static async search(
        query: string = '',
        workbook: string = '',
        project: string = '',

    ): Promise<any> {
        try {
            const response = await apiClient.get(

// ============================================
// PERMITTED VIEWS
// ============================================
export const getPermittedViews = async (oidcSub: string): Promise<PermittedViewsResponse> => {
    try {
        if (!oidcSub) {
            console.error('No oidc_sub provided');
            return {
                success: false,
                error: 'User not authenticated - oidc_sub missing',
                views: [],
                total_count: 0,
                sites_scanned: 0,
            };
        }

        const csrfToken = getCsrfToken();

        const response = await fetch(`${API_BASE_URL}/api/tableau/permitted-views/`, {
            method: 'POST',
            credentials: 'include',
            headers: {
                'Content-Type': 'application/json',
                'X-CSRFToken': csrfToken,
            },
            body: JSON.stringify({ oidc_sub: oidcSub }),
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const data = await response.json();
        return data;
    } catch (error: any) {
        console.error('Error fetching permitted views:', error);
        return {
            success: false,
            error: error.message || 'Failed to fetch permitted views',
            views: [],
            total_count: 0,
            sites_scanned: 0,
        };
    }
};

export const getPermittedDashboards = async (): Promise<PermittedViewsResponse> => {
    try {
        const oidcSub = localStorage.getItem('oidc_sub');
        if (!oidcSub) {
            console.error('No oidc_sub found in localStorage');
            return {
                success: false,
                error: 'User not authenticated - oidc_sub missing',
                views: [],

                sites_scanned: 0,

        const response = await fetch(`${API_BASE_URL}/api/dashboards/`, {
            method: 'POST',
            credentials: 'include',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ oidc_sub: oidcSub }),
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const data = await response.json();
        return data;
    } catch (error: any) {
        console.error('Error fetching permitted dashboards:', error);
        return {
            success: false,
            error: error.message || 'Failed to fetch permitted dashboards',
            views: [],
            total_count: 0,
            sites_scanned: 0,
        };
    }
};

export const pinCard = async (card: InsightCard): Promise<PinCardResponse> => {
    try {
        const response = await fetch('/api/pinned-cards/pin/', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'include',
            body: JSON.stringify({
                card_id: card.id,
                customized_name: card.customized_name || '',
                url_id: card.url_id || '',
                view_name: card.view_name,
                view_repository_url: card.view_repository_url || '',
                view_index: card.view_index || 0,
                workbook_name: card.workbook_name,
                workbook_repo_url: card.workbook_repo_url || '',
                site_name: card.site_name || '',
                last_accessed: card.last_accessed || '',
                is_public: card.is_public || false,
                url_attempt_1_url_id: card.url_attempt_1_url_id || '',
                url_attempt_2_repo: card.url_attempt_2_repo,
            }),
        });
        return await response.json();
    } catch (error: any) {
        console.error('Error pinning card:', error);
        if (error.response?.status === 400) {
            throw new Error('Card is already pinned');
        }
        throw error;
    }
};

export const unpinCard = async (cardId: number): Promise<PinCardResponse> => {
    try {
        const response = await fetch(`/api/pinned-cards/unpin/${cardId}/`, {
            method: 'DELETE',
            credentials: 'include',
        });
        return await response.json();
    } catch (error: any) {
        console.error('Error unpinning card:', error);
        if (error.response?.status === 404) {
            throw new Error('Card is not pinned');
        }
        throw error;
    }
};
